// obj_pause_menu_step.txt
// Get inputs
var up_key = keyboard_check_pressed(ord("W")) || keyboard_check_pressed(vk_up);
var down_key = keyboard_check_pressed(ord("S")) || keyboard_check_pressed(vk_down);
var accept_key = keyboard_check_pressed(vk_enter);

// Store number of options in the current menu
op_length = array_length(option[menu_level]);

// Move through the menu
pos += down_key - up_key;
if (pos >= op_length) { pos = 0; }
if (pos < 0) { pos = op_length - 1; }

// Using the option
if (accept_key) {
    var _sml = menu_level;

    switch (menu_level) {
        // Pause menu
        case 0:
            switch (pos) {
                case 0: 
                    game_paused = false;
                    instance_destroy(); 
                    break; // Back to Game
                case 1: 
                    menu_level = 1; 
                    break; // Settings
                case 2: 
                    game_end(); 
                    break; // Quit Game
            }
            break;

        // Settings menu
        case 1:
            switch (pos) {
                case 0: // Window Size
                    break;
                case 1: // Brightness
                    break;
                case 2: // Controls
                    break;
                case 3: 
                    instance_create_layer(x, y, "UI", obj_skinpicker); 
                    break; // Change Skin
                case 4: 
                    instance_create_layer(x, y, "UI", obj_color_wheel); 
                    break; // Change Color
                case 5: 
                    menu_level = 0; 
                    break; // Back
            }
            break;
    }

    // Reset position if the menu level has changed
    if (_sml != menu_level) { pos = 0; }
    op_length = array_length(option[menu_level]);
}

// Ensure the game is paused
if (game_paused) {
    // Pause game logic here
}
