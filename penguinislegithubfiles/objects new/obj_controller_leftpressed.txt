if (mouse_check_button_pressed(mb_left)) {
    // Log the input detection
    show_debug_message("DEBUG: Mouse Left Press Detected in obj_controller.");

    // Skip input if chat is active
    if (global.chat_active) {
        show_debug_message("DEBUG: Input ignored because global.chat_active is true.");
        return; // Exit early to avoid conflict
    }

    // Create the skin picker if not already present
    if (!instance_exists(obj_skinpicker)) {
        if (instance_exists(global.player_instance)) {
            instance_create_layer(global.player_instance.x, global.player_instance.y, "Instances", obj_skinpicker);
        } else {
            instance_create_layer(view_xview[0] + view_wview[0] / 2, view_yview[0] + view_hview[0] / 2, "Instances", obj_skinpicker);
        }
    }
}
