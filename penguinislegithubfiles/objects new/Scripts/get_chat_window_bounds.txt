function get_chat_window_bounds() {
    var camera_x = camera_get_view_x(global.camera);
    var camera_y = camera_get_view_y(global.camera);
    var camera_width = camera_get_view_width(global.camera);
    var camera_height = camera_get_view_height(global.camera);

    // Chat window dimensions
    var menu_width = 300; // Adjust width
    var menu_height = 120; // Adjust height
    var menu_x = camera_x + (camera_width - menu_width) / 2; // Center horizontally
    var menu_y = camera_y + camera_height - menu_height - 20; // Position at bottom

    // Close button dimensions
    var close_button_size = 20;
    var close_x = menu_x + menu_width - close_button_size - 10;
    var close_y = menu_y + 10;

    return {
        menu_x: menu_x,
        menu_y: menu_y,
        menu_width: menu_width,
        menu_height: menu_height,
        close_x: close_x,
        close_y: close_y,
        close_button_size: close_button_size,
    };
}
